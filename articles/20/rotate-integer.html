(define (reverse-integer n)
  (let loop ((n n) (result 0))
    (if (< n 1)
        result
        (loop (quotient n 10)
              (+ (* result 10)
                 (remainder n 10))))))

(define (rotate-integer n)
  (let loop ((n n)
            (multiplier 10)
            (result 0))
    (if (< n 10)
        (+ result (remainder n 10))
        (loop (quotient n 10)
              (* multiplier 10)
              (+ result (* multiplier (remainder n 10)))))))
